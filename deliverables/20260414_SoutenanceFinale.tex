% ======================================================================
% SLIDE 3: LOB AND SPOOFING
% ======================================================================
\begin{frame}{Spoofing in the Limit Order Book}
    \begin{block}{Mechanism}
        \begin{itemize}
            \item Non-bona fide orders placed to create false supply/demand pressure
            \item Mid-price shifts towards the manipulator's genuine order
            \item Spoof orders cancelled once the genuine order is filled
        \end{itemize}
    \end{block}
    \begin{alertblock}{Detection Challenges}
        \begin{enumerate}
            \item \textbf{Contamination:} training data contains unknown manipulations
            \item \textbf{No ground truth:} no ``manipulated / not manipulated'' labels
            \item \textbf{Non-stationarity:} market regime changes (volatility, liquidity)
        \end{enumerate}
    \end{alertblock}
    \note{Spoofing is prohibited under EU MAR and the US Dodd-Frank Act, yet detection remains difficult because the same order flow patterns arise from legitimate trading strategies. The absence of labels forces an unsupervised, score-based approach.}
\end{frame}

% ======================================================================
% SLIDE 4: WHY PROBABILISTIC, SCORE-BASED DETECTION
% ======================================================================
\begin{frame}{Why Probabilistic, Score-Based Detection}
    Binary classification is unsuitable for three reasons:
    \begin{enumerate}
        \item No ground-truth labels exist at the order level in equity markets
        \item Manipulation is a spectrum of intent, not a discrete category
        \item A continuous score allows ranking events by severity and calibrating thresholds to a desired false discovery rate
    \end{enumerate}

    \vspace{0.3cm}
    \textbf{Research questions:}
    \begin{itemize}
        \item Can unsupervised models detect anomalous order flow without labels?
        \item Do different models agree on the same discriminative features?
        \item Can spoofing profitability be estimated directly from model predictions?
        \item Do detection patterns remain stable across years?
    \end{itemize}
    \note{The probabilistic formulation enables both ranking-based surveillance and economic interpretability through the spoofing gain framework of Fabre and Challet.}
\end{frame}

\begin{frame}{Feature Engineering: 89 Features in 7 Groups}
    \begin{table}[h]
    \centering
    \footnotesize
    \begin{tabular}{lrl}
    \toprule
    \textbf{Group} & \textbf{Count} & \textbf{Key quantities} \\
    \midrule

    Price dynamics     & 13 & Mid-price, spread, log return, depth ratio \\
    Elasticity         & 4  & Sweep cost ($Q{=}1000$), book slope ($D{=}5$) \\
    Volatility         & 6  & Rolling $\sigma$ ($W{=}50,100$), price range, $\Delta t$ \\
    Event flow         & 18 & Cancel/trade size \& rapidity \cite{poutre_2024} \\
    Hawkes memory      & 38 & Spatial + temporal decay, $4\eta \times 3\beta$ \cite{fabre_2025} \\
    OFI                & 5  & Multi-level order flow imbalance (L1--L5) \\
    \midrule
    \textbf{Total}     & \textbf{89} & ${\sim}7{\times}$ larger than Poutr{\'e} et al.\ (${\sim}13$) \\
    \bottomrule
    \end{tabular}
    \end{table}
    \begin{itemize}
        \item Preprocessing: replace inf/NaN, trim 3{,}000 warmup rows, clip at $[0.1\%, 99.9\%]$ quantiles
        \item Scaling: Empirical Box-Cox + z-score, fitted on day~1 only
    \end{itemize}
    \note{The feature library integrates contributions from three reference papers. Hawkes memory features capture self-exciting order flow patterns with spatial decay, absent from Poutr\'e et al.'s Level-1 feature set.}
\end{frame}

% ======================================================================
% SLIDE 7: MODEL 1 --- TRANSFORMER + OC-SVM
% ======================================================================
\begin{frame}{Model 1: Transformer Bottleneck AE + OC-SVM \cite{poutre_2024}}
    \begin{block}{Architecture}
        \begin{itemize}
            \item Bottleneck Transformer AE: input $(25 \times 89)$ $\to$ 128-dim latent $\to$ reconstruction
            \item 6 layers, 8 heads, $d_{\text{ff}}{=}512$; trained with MSE loss (Adam, $\eta{=}10^{-4}$, patience 20)
        \end{itemize}
    \end{block}
    \begin{block}{OC-SVM in Latent Space}
        \begin{itemize}
            \item Nystr{\"o}m approximation ($m{=}300$ landmarks), RBF $\gamma$ via median heuristic
            \item SGD training (500 epochs, $\nu{=}0.01$), fitted once after all training days
        \end{itemize}
    \end{block}
    \begin{block}{Anomaly Score}
        Negated OC-SVM decision function $d(\mathbf{z})$; higher $\to$ more anomalous
    \end{block}
    \note{The Transformer captures complex multivariate dependencies across 25-event windows. The OC-SVM boundary in the 128-dim bottleneck space provides a density-aware decision function that complements the reconstruction error.}
\end{frame}

% ======================================================================
% SLIDE 8: MODEL 2 --- PNN
% ======================================================================
\begin{frame}{Model 2: Probabilistic Neural Network \cite{fabre_2025}}
    \begin{block}{Architecture}
        \begin{itemize}
            \item Single hidden layer ($d_h{=}64$, ReLU), Markovian: last time step only
            \item Output: skew-normal parameters $(\mu, \sigma, \alpha)$ for $\Delta p_{\text{mid}}$
        \end{itemize}
    \end{block}
    \begin{block}{Training Objective}
        Skew-normal NLL: $\mathcal{L} = -\frac{1}{|\mathcal{B}|}\sum_i \ln\bigl(\frac{2}{\sigma_i}\,\varphi(z_i)\,\Phi(\alpha_i z_i) + 10^{-10}\bigr)$
    \end{block}
    \begin{block}{Anomaly Score}
        Spoofing gain $G_t = \mathbb{E}[C_{\text{normal}}] - \mathbb{E}[C_{\text{spoof}}]$; flagged if $G_t > 0$
    \end{block}
    \note{The PNN is the only model with a direct economic interpretation. The spoofing gain quantifies the expected profit from a hypothetical spoofing strategy, using parameters $Q{=}4500$, $q{=}100$, $\delta_b{=}0.01$, maker fee 0, taker fee 0.0008 (Euronext Paris large-cap schedule).}
\end{frame}

% ======================================================================
% SLIDE 9: MODEL 3 --- PRAE
% ======================================================================
\begin{frame}{Model 3: Probabilistic Robust Autoencoder \cite{lindenbaum2022prae}}
    \begin{block}{Architecture}
        \begin{itemize}
            \item Same Transformer backbone as TF-OCSVM
            \item Learnable per-sample gate $g_i = \min\bigl(\max(\mu_i + \varepsilon_i,\; 0),\; 1\bigr)$, \quad $\varepsilon_i \sim \mathcal{N}(0,\, 0.5^2)$
        \end{itemize}
    \end{block}
    \begin{block}{Training Objective}
        $\mathcal{L}_1 = \mathbb{E}\bigl(\sum_i g_i \|x_i - \hat{x}_i\|^2 - \lambda \|\mathbf{g}\|_1\bigr)$; $\lambda$ chosen via grid search on day~1
    \end{block}
    \begin{block}{Anomaly Score}
        Reconstruction error $s_t = \|S_t - \hat{S}_t\|^2$; thresholded by RFDR ($w{=}500$, $\alpha{=}0.05$)
    \end{block}
    \note{The gating mechanism is designed to downweight contaminated training samples. In practice, the gate parameters collapsed to the initialisation value on both periods, so the PRAE effectively operated as a standard Transformer AE.}
\end{frame}

% ======================================================================
% SLIDE 10: THRESHOLDING METHODS
% ======================================================================
\begin{frame}{Anomaly Thresholding: Four Label-Free Methods}
    \begin{table}[h]
    \centering
    \footnotesize
    \begin{tabular}{lp{6.5cm}}
    \toprule
    \textbf{Method} & \textbf{Mechanism} \\
    \midrule
    POT  & Batch EVT: fits GPD to exceedances above 98th percentile \\
    SPOT \cite{siffer:hal-01640325} & Streaming EVT: updates GPD parameters as new peaks arrive \\
    DSPOT \cite{siffer:hal-01640325} & Drift-aware SPOT: adds moving-average correction for non-stationarity \\
    RFDR & Rolling Benjamini-Hochberg: controls false discovery rate at $\alpha$ on a sliding window \\
    \bottomrule
    \end{tabular}
    \end{table}
    \begin{itemize}
        \item TF-OCSVM also uses a baseline $\tau_0 = Q_{1-\nu}$ (percentile of training dissimilarity scores)
        \item PNN uses an economic threshold: $G_t > 0$ (spoofing profitable)
    \end{itemize}
    \note{DSPOT degenerates on PNN bounded scores (flags 99.8\%). RFDR and POT are the most stable label-free methods across all three score streams.}
\end{frame}

\section{Discussion and Conclusion}
\subsection{Discussion}

% ======================================================================
% SLIDE 32: LIMITATIONS
% ======================================================================
\begin{frame}{Limitations}
    \begin{enumerate}
        \item No ground-truth manipulation labels: precision/recall cannot be computed
        \item Single instrument (TOTF.PA), only 3 distal test days per split
        \item Nystr{\"o}m $m{=}300$ landmarks: unquantified approximation error
        \item Feature parameters adopted from crypto literature without equity-specific ablation
        \item Box-Cox scaler fitted on day~1 only: potential miscalibration under shift
        \item Training restricted to first hour: misses intraday regime changes
        \item PNN single hidden layer: limited capacity for complex distributions
        \item RFDR Gaussian assumption violated by heavy-tailed scores
        \item SPOT/DSPOT $n_{\text{init}}{=}1000$: insufficient tail mass in low-density periods
    \end{enumerate}
    \note{The most consequential limitation is the absence of labels, which prevents reporting standard classification metrics and forces all evaluation to rely on indirect proxies (gain alignment, jump co-occurrence, cross-year overlap).}
\end{frame}

% ======================================================================
% SLIDE 33: COMPARISON OF APPROACHES
% ======================================================================

        \item TF-OCSVM: emergent gain alignment, but collapses on cross-year data
        \item PRAE: most robust ($J{=}0.18$), but flags less spoofable states
    \end{itemize}
    \note{No single model dominates. A production system should combine PRAE for robustness with PNN for economic interpretability, using RFDR thresholding.}
\end{frame}

% ======================================================================
% SLIDE 34: INTERPRETATION
% ======================================================================
\begin{frame}{Interpretation: Detected Anomalies and Model Blind Spots}
    \textbf{Spoofing/layering signatures:}
    \begin{itemize}
        \item Elevated Hawkes limit-order memory (large, deep placements)
        \item High cancellation rapidity at L1; maps to place $\to$ shift $\to$ execute $\to$ cancel
    \end{itemize}
    \vspace{0.15cm}
    \textbf{Legitimate events captured:}
    \begin{itemize}
        \item Volume surges (block trades, institutional execution)
        \item Spread expansions (news impact, low-liquidity episodes)
    \end{itemize}
    \vspace{0.15cm}
    \textbf{Likely missed patterns:}
    \begin{itemize}
        \item Subtle, low-volume spoofing (below anomaly score thresholds)
        \item Multi-day strategic manipulation (single-day windows)
        \item Quote stuffing via message-rate flooding (not in Level-2 data)
    \end{itemize}
    \takehome{Definitive classification requires regulatory cross-referencing; the pipeline identifies candidate events with structural spoofing properties.}
    \note{The two-cluster taxonomy from HDBSCAN partially separates the spoofing-type pattern from the volume-type pattern. Full disambiguation requires order-level (Level-3) data.}
\end{frame}

% SECTION 12: CONCLUSION
% ######################################################################
\subsection{Conclusion}

% ======================================================================
% SLIDE 35: CONTRIBUTIONS
% ======================================================================
\begin{frame}{Contributions}
    \begin{enumerate}
        \item Modular unsupervised pipeline combining three complementary models (TF-OCSVM, PNN, PRAE) on a single equity LOB dataset (TOTF.PA, 2010/2015/2017)
        \item Feature library of 89 features (${\sim}7\times$ Poutr{\'e} et al.); Hawkes limit-order memory and cancellation features identified as top drivers across all models
        \item First application of the spoofing gain framework (Fabre \& Challet) to equity LOB data: 0.07\% of 2017 windows yield $G > 0$
        \item Systematic comparison of four label-free thresholds: RFDR and POT are the most stable; DSPOT degenerates on bounded scores
        \item Robustness hierarchy established: PRAE > TF-OCSVM > PNN (Jaccard 0.07--0.18 vs.\ ${\sim}0$)
    \end{enumerate}
    \note{The main finding is that unsupervised probabilistic models can detect structurally consistent anomalies across years, but each model captures a different anomaly facet. The PRAE is the recommended single-model choice for robustness.}
\end{frame}

% ======================================================================
% SLIDE 36: FUTURE WORK
% ======================================================================
\begin{frame}{Future Work}
    \begin{enumerate}
        \item \textbf{Multi-asset evaluation:} extend to diverse instruments, venues, and asset classes to test generalisability beyond TOTF.PA
        \item \textbf{Level-3 integration:} use message-level data for granular event classification and direct observation of order placement/cancellation intent
        \item \textbf{Deeper density estimation:} replace PNN single-layer MLP with attention-based or normalising-flow architectures for richer conditional distributions
        \item \textbf{Online continual learning:} transition from batch sequential training to online parameter updates with automated concept drift detection
        \item \textbf{Cryptocurrency validation:} evaluate on crypto LOBs where manipulation prevalence is higher, using the same pipeline and features
        \item \textbf{Synthetic calibration:} calibrate thresholds against spoofing patterns generated by agent-based market simulators
    \end{enumerate}
    \note{The most impactful next step is cryptocurrency validation, where the higher manipulation base rate provides a natural validation set for the pipeline's detection capabilities.}
\end{frame}
